CREATE TABLE DEPARTMENT (
    DEPARTMENT_NAME VARCHAR(25) PRIMARY KEY
);



CREATE TABLE  USERS (
     FIRST_NAME varchar(255),
    LAST_NAME varchar(255),
   EMPLOYEE_ID int PRIMARY KEY,
   
    DEPARTMENT_NAME VARCHAR(25),
Foreign Key (DEPARTMENT_NAME)
REFERENCES department(DEPARTMENT_NAME)
);


CREATE TABLE CARTRIDGE (
   CARTRIDGE_TYPE varchar(255) PRIMARY KEY,
   QUANTITY INT
);


CREATE TABLE CARTRIDGE_USED_IN_DEPARTMENT(
      DEPARTMENT_NAME VARCHAR(25) ,
      CARTRIDGE_TYPE varchar(255) ,
primary key( DEPARTMENT_NAME,CARTRIDGE_TYPE ),
Foreign Key (DEPARTMENT_NAME)
REFERENCES  DEPARTMENT(DEPARTMENT_NAME),
Foreign Key (CARTRIDGE_TYPE)
REFERENCES CARTRIDGE(CARTRIDGE_TYPE)
);



CREATE TABLE TRANSACTIONS( 
    RECEIPT_NUMBER INT PRIMARY KEY,
    EMPLOYEE_ID int,
    LAST_NAME varchar(255),
    FIRST_TNAME varchar(255),
    CARTRIDGE_TYPE varchar(255),
    DEPARTMENT_NAME VARCHAR(25), 
    QUANTITY_PURCHASED INT, 
    DATE_OF_PURCHASE DATE  ,


Foreign Key (EMPLOYEE_ID)
REFERENCES USERS (EMPLOYEE_ID),


Foreign Key (DEPARTMENT_NAME)
REFERENCES department (DEPARTMENT_name),


Foreign Key (CARTRIDGE_TYPE)
REFERENCES CARTRIDGE (CARTRIDGE_TYPE)

);
